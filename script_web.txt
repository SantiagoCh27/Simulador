// Función para hacer clic en un elemento
const clickEnElemento = (elemento, delay = 0) => {
    setTimeout(() => {
        if (elemento) elemento.click();
    }, delay);
};

// Función para eliminar el texto y escribir un "1"
const escribirUno = (selector, delay = 0) => {
    setTimeout(() => {
        const input = document.querySelector(selector);
        if (input) {
            input.value = ""; // Borra el contenido actual
            input.dispatchEvent(new Event("input", { bubbles: true })); // Simula que el usuario borró

            setTimeout(() => {
                input.value = "1"; // Escribe el "1"
                input.dispatchEvent(new Event("input", { bubbles: true })); // Simula la escritura
            }, 100); // Pequeño delay para hacer más realista
        }
    }, delay);
};

// ⏱ Ejecución inmediata con delays personalizables
const botonPlay = document.querySelectorAll(".kHtcsd")[12]; // Cambia el índice si es necesario
const botonPlay2 = document.querySelectorAll(".kHtcsd")[2]; // Cambia el índice si es necesario
const botonPlay3 = document.querySelectorAll(".kHtcsd")[13]; // Cambia el índice si es necesario
clickEnElemento(botonPlay, 0);  // Clic en el botón "Play"

escribirUno("#APjFqb", 200);  // Espera 200ms, borra el contenido y escribe "1"
clickEnElemento(botonPlay2, 400);  // Clic en el botón "Mute" y Espera 400ms
clickEnElemento(botonPlay3, 600);  // Clic en el botón "Stop" y Espera 600ms
